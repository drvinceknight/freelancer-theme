\begin{frame}
    \begin{align*}
        I_n &= \text{random}  && \text{Inter arrival time}\\
        A_n &= I_{n-1} + I_n  && \text{Arrival date}\\
        T_n &= \text{random}  && \text{Service time}\\
        S_n &= \max(T_n, E_{n + 1}) && \text{Service start date}\\
        E_n &= S_n + T_n && \text{Service end date}\\
    \end{align*}
\end{frame}
